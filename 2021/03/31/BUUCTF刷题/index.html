#!/usr/bin/env python

from PIL import Image


def dtob(num):
    temp = bin(num).replace("0b", "")
    length = len(temp)
    over = length // 8
    temp = temp.zfill((over + 1) * 8)
    return temp


def getSecret(mpath):
    secret = ""
    file = open(mpath, "rb")
    content = file.read()
    length = len(content)
    for i in range(length):
        tmp = content[i]
        tmp = dtob(tmp)
        secret += tmp
    return secret
